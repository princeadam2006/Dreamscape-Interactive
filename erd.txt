Table users {
  id bigint [pk, increment]
  name varchar(255)
  username varchar(255) [unique]
  email varchar(255) [unique]
  email_verified_at timestamp [null]
  password varchar(255)
  remember_token varchar(100) [null]
  notification_preferences json [null]
  created_at timestamp
  updated_at timestamp
}

Table roles {
  id bigint [pk, increment]
  name varchar(255) [unique]
  guard_name varchar(255)
  created_at timestamp
  updated_at timestamp
}

Table permissions {
  id bigint [pk, increment]
  name varchar(255) [unique]
  guard_name varchar(255)
  created_at timestamp
  updated_at timestamp
}

Table model_has_roles {
  role_id bigint
  model_type varchar(255)
  model_id bigint

  indexes {
    (model_id, model_type)
    (role_id)
  }
}

Table model_has_permissions {
  permission_id bigint
  model_type varchar(255)
  model_id bigint

  indexes {
    (model_id, model_type)
    (permission_id)
  }
}

Table role_has_permissions {
  permission_id bigint
  role_id bigint

  indexes {
    (role_id)
    (permission_id)
  }
}

Table items {
  id bigint [pk, increment]
  name varchar(255)
  description text
  type varchar(50)
  rarity varchar(50)
  required_level int [null]
  power smallint
  speed smallint
  durability smallint
  magical_properties text
  tradeable_default boolean
  created_at timestamp
  updated_at timestamp
}

Table inventory_items {
  id bigint [pk, increment]
  user_id bigint
  item_id bigint
  locked boolean
  created_at timestamp
  updated_at timestamp
}

Table trades {
  id bigint [pk, increment]
  initiator_user_id bigint
  receiver_user_id bigint
  status varchar(50)
  message text [null]
  created_at timestamp
  updated_at timestamp
}

Table trade_items {
  id bigint [pk, increment]
  trade_id bigint
  inventory_item_id bigint
  user_id bigint
  role_in_trade varchar(50)
}

Table notifications {
  id bigint [pk, increment]
  user_id bigint
  type varchar(255)
  data json
  read_at timestamp [null]
  created_at timestamp
  updated_at timestamp
}

Table audit_logs {
  id bigint [pk, increment]
  user_id bigint
  action varchar(100)
  target_user_id bigint [null]
  target_item_id bigint [null]
  meta json [null]
  created_at timestamp
}

Ref: model_has_roles.role_id > roles.id
Ref: model_has_roles.model_id > users.id

Ref: model_has_permissions.permission_id > permissions.id
Ref: model_has_permissions.model_id > users.id

Ref: role_has_permissions.role_id > roles.id
Ref: role_has_permissions.permission_id > permissions.id

Ref: inventory_items.user_id > users.id
Ref: inventory_items.item_id > items.id

Ref: trades.initiator_user_id > users.id
Ref: trades.receiver_user_id > users.id

Ref: trade_items.trade_id > trades.id
Ref: trade_items.inventory_item_id > inventory_items.id
Ref: trade_items.user_id > users.id

Ref: notifications.user_id > users.id

Ref: audit_logs.user_id > users.id
Ref: audit_logs.target_user_id > users.id
Ref: audit_logs.target_item_id > items.id
